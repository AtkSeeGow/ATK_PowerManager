Ubuntu

設定裝置名稱

sudo hostnamectl set-hostname {{ '{' }}裝置名稱{{ '}' }}

設定無線網路

sudo cp /usr/share/doc/netplan/examples/wireless.yaml /etc/netplan
sudo nano /etc/netplan/wireless.yaml 
---------------------------------------
network:
  version: 2
  renderer: networkd
  wifis:
    wlan0:
      dhcp4: yes
      dhcp6: no
      access-points:
        "TP-Link A608":
          password: "*****************"
---------------------------------------
sudo reboot

設定資料夾共享

sudo apt-get install samba
sudo usermod -a -G sambashare {{ '{' }}使用者名稱{{ '}' }}
sudo pdbedit -a -u {{ '{' }}使用者名稱{{ '}' }}
sudo nano /etc/samba/smb.conf
---------------------------------------
[{{ '{' }}使用者名稱{{ '}' }}]
  comment = {{ '{' }}使用者名稱{{ '}' }}'s home
  path = /home/{{ '{' }}使用者名稱{{ '}' }}
  read only = no
  guest ok = no
  browseable = yes
  create mask = 0644
  directory mask = 0755
---------------------------------------
sudo reboot

偵測網路斷線並重啟

nano check_connection.sh
---------------------------------------
#!/bin/bash

str1="$(sudo iwconfig)"
str2="TP-Link"
str3=$(echo $str1 | grep "${{ '{' }}str2{{ '}' }}")

if [ "$str3" = "" ]
then
  echo $(date) >> reboot.txt
  sudo reboot
fi

echo $(date) >> iwconfig.txt
---------------------------------------
